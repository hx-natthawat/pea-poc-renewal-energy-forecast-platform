   ```sql
   -- Consumer data (300,000 rows)
   CREATE TABLE consumers (
       consumer_id BIGINT PRIMARY KEY,
       transformer_id INT,
       phase CHAR(1),
       capacity_kw DECIMAL
   );

       voltage DECIMAL,
       power DECIMAL

          AVG(voltage) as avg_voltage,

   FROM meter_data m
   JOIN consumers c ON m.consumer_id = c.consumer_id
   GROUP BY 1, 2;
   ```
